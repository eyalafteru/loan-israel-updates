```javascript
// דוגמה לנוסחת PMT להחזר חודשי
function calculateMonthlyPayment(principal, rate, months) {
    const monthlyRate = rate / 100 / 12;
    const payment = principal * (monthlyRate * Math.pow(1 + monthlyRate, months)) / 
                   (Math.pow(1 + monthlyRate, months) - 1);
    return Math.round(payment);
}

// חישוב סה"כ ריבית
totalInterest = (monthlyPayment * months) - principal;
```

- Memoization: לחישובים חוזרים

### ⚡ Performance Optimization - קוד קל משקל

#### 1. Lazy Loading מתקדם
```javascript
// לא רק תמונות - גם רכיבים כבדים
const lazyObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting && entry.target.dataset.lazy) {
            loadComponent(entry.target);
            lazyObserver.unobserve(entry.target);
        }
    });
}, { threshold: 0.1, rootMargin: '50px' });
```

#### 2. CSS Containment
```css
/* מונע reflow/repaint בכל הדף */
.wpc-[topic]-[4random]-calculator,
.wpc-[topic]-[4random]-widget {
    contain: layout style paint !important;
}
```

#### 3. Event Delegation מרכזי
```javascript
// במקום 20 listeners - אחד בלבד
const mainContainer = document.getElementById('wpc-[topic]-[4random]-main');
mainContainer.addEventListener('click', function(e) {
    const action = e.target.closest('[data-action]');
    if (action && handlers[action.dataset.action]) {
        handlers[action.dataset.action](e);
    }
}, { passive: true });
```

#### 4. Memory Cleanup Pattern
```javascript
window.WPC_[Topic]_[Random].destroy = function() {
    // Disconnect observers
    if (lazyObserver) lazyObserver.disconnect();
    // Remove listeners
    mainContainer.removeEventListener('click', delegatedHandler);
    // Clear intervals/timeouts
    clearInterval(updateInterval);
    // Nullify references
    heavyObjects = null;
};
```

#### 5. Font Loading Strategy
```css
/* בתחילת ה-CSS */
@font-face {
    font-family: 'Assistant';
    font-display: swap; /* מונע FOIT */
    unicode-range: U+0590-05FF, U+FB1D-FB4F; /* רק עברית */
}
```

## FAQ JavaScript בטוח לוורדפרס עם תגיות SEO

```javascript
(function() {
    'use strict';
    
    // 1. בדיקת התנגשויות
    const NS = 'WPC_[Topic]_[Random]';
    if (window[NS]) return;
    
    // 2. קונטיינר בטוח
    const container = document.getElementById('wpc-[topic]-[4random]-main');
    if (!container) return;
    
    // 3. State מרכזי
    const state = {
        loaded: new Set(),
        listeners: new Map()
    };
    
    // 4. Lazy Loading Pattern
    const componentLoader = {
        observer: null,
        
        init() {
            this.observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting && !state.loaded.has(entry.target.id)) {
                        this.load(entry.target);
                    }
                });
            }, { threshold: 0.1 });
            
            container.querySelectorAll('[data-lazy]').forEach(el => {
                this.observer.observe(el);
            });
        },
        
        load(element) {
            const type = element.dataset.lazy;
            switch(type) {
                case 'calculator':
                    // טעינה דינמית של מחשבון
                    this.loadCalculator(element);
                    state.loaded.add(element.id);
                    this.observer.unobserve(element);
                    break;
            }
        },
        
        loadCalculator(element) {
            // הפעלת המחשבון
            console.log('טוען מחשבון...');
        }
    };
    
    // 5. Event Delegation בטוח
    const eventHandler = {
        init() {
            // רק listener אחד!
            container.addEventListener('click', this.handleClick.bind(this), { passive: true });
            
            // Touch events למובייל
            if ('ontouchstart' in window) {
                container.addEventListener('touchstart', this.handleTouch.bind(this), { passive: true });
            }
        },
        
        handleClick(e) {
            const target = e.target.closest('[data-action]');
            if (!target) return;
            
            e.preventDefault();
            const action = target.dataset.action;
            
            // Route to specific handlers
            if (this.actions[action]) {
                this.actions[action](target, e);
            }
        },
        
        actions: {
            'calculate': (el) => console.log('חישוב...'),
            'toggle-faq': (el) => console.log('פתיחת FAQ...')
        }
    };
    
    // 6. Cleanup מובנה
    const cleanup = {
        destroy() {
            // Observers
            if (componentLoader.observer) {
                componentLoader.observer.disconnect();
            }
            
            // Event listeners
            container.removeEventListener('click', eventHandler.handleClick);
            
            // Clear state
            state.loaded.clear();
            state.listeners.clear();
            
            // Remove from window
            delete window[NS];
        }
    };
    
    // 7. Init with performance mark
    performance.mark('wpc-init-start');
    
    window[NS] = {
        init() {
            componentLoader.init();
            eventHandler.init();
        },
        destroy: cleanup.destroy.bind(cleanup)
    };
    
    // Auto-init on DOM ready
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', window[NS].init);
    } else {
        window[NS].init();
    }
    
    performance.mark('wpc-init-end');
    performance.measure('wpc-init', 'wpc-init-start', 'wpc-init-end');
    
})();
```

## ⚡ כללי הזהב לקוד מהיר ובטוח בוורדפרס:

1. **"Isolate Everything"** - בידוד מוחלט מוורדפרס (prefix + namespace + container)
2. **"Load on Visibility"** - רכיבים כבדים נטענים רק כשנראים (Intersection Observer)
3. **"One Listener Rules All"** - Event delegation תמיד (מקסימום 10 listeners בכל העמוד)

### מחשבון החזר הלוואה (PMT):
```javascript
function calculatePMT(principal, rate, periods) {
    const monthlyRate = rate / 100 / 12;
    const payment = principal * (monthlyRate * Math.pow(1 + monthlyRate, periods)) / 
                   (Math.pow(1 + monthlyRate, periods) - 1);
    return Math.round(payment);
}
```

### מחשבון ריבית פשוטה:
```javascript
function simpleInterest(principal, rate, time) {
    return principal * (rate / 100) * time;
}
```

### מחשבון ריבית דריבית:
```javascript
function compoundInterest(principal, rate, time, compound) {
    return principal * Math.pow(1 + (rate / 100) / compound, compound * time) - principal;
}
```

### מחשבון יכולת החזר:
```javascript
function affordabilityRatio(monthlyIncome, monthlyExpenses) {
    const availableIncome = monthlyIncome - monthlyExpenses;
    const maxPayment = availableIncome * 0.3; // 30% rule
    return Math.floor(maxPayment);
}
```
