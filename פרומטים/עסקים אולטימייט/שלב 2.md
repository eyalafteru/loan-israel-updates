# 📋 חלק א: פרוטוקול SEO

## Schema Markup נדרש

### 1. Person - אייל עובדיה (חובה!)

```json
<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@type": "Person",
  "name": "אייל עובדיה",
  "jobTitle": "מנהל מקצועי ויועץ ראשי",
  "worksFor": {
    "@type": "Organization",
    "name": "רק תבקש - אפטריו בע״מ",
    "url": "https://loan-israel.co.il"
  },
  "description": "בעל ניסיון עשיר בתחום ההלוואות לעסקים",
  "knowsAbout": ["הלוואות לעסקים", "קרן בערבות מדינה", "מימון עסקי"],
  "telephone": "+972-53-428-8957"
}
</script>
```

### 2. Organization (חובה!)

```json
<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@type": "Organization",
  "name": "רק תבקש - אפטריו בע״מ",
  "url": "https://loan-israel.co.il",
  "telephone": "+972-53-428-8957",
  "address": {
    "@type": "PostalAddress",
    "addressCountry": "IL"
  }
}
</script>
```

### 3. BreadcrumbList (חובה!)

```json
<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@type": "BreadcrumbList",
  "itemListElement": [{
    "@type": "ListItem",
    "position": 1,
    "name": "דף הבית",
    "item": "https://loan-israel.co.il/"
  },{
    "@type": "ListItem",
    "position": 2,
    "name": "הלוואות לעסקים",
    "item": "https://loan-israel.co.il/הלוואות-לעסקים/"
  },{
    "@type": "ListItem",
    "position": 3,
    "name": "[שם העמוד]",
    "item": "[URL מלא]"
  }]
}
</script>
```

### 4. FAQPage (אם יש FAQ)

```json
<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@type": "FAQPage",
  "mainEntity": [{
    "@type": "Question",
    "name": "[שאלה 1]",
    "acceptedAnswer": {
      "@type": "Answer",
      "text": "[תשובה 1]"
    }
  }]
}
</script>
```

### 5. FinancialProduct (לעמודי מוצר)

```json
<script type="application/ld+json">
{
  "@context": "https://schema.org",
  "@type": "FinancialProduct",
  "name": "[שם המוצר]",
  "description": "[תיאור קצר]",
  "annualPercentageRate": {
    "@type": "QuantitativeValue",
    "minValue": "[ריבית מינימום]",
    "maxValue": "[ריבית מקסימום]",
    "unitText": "%"
  }
}
</script>
```
